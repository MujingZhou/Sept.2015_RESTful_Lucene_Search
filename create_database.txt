DROP DATABASE IF EXISTS Publication;
CREATE DATABASE IF NOT EXISTS Publication;
USE Publication;
CREATE TABLE IF NOT EXISTS Publication.Author(authorId INTEGER not NULL AUTO_INCREMENT,authorName VARCHAR(255),PRIMARY KEY (authorId));
CREATE TABLE IF NOT EXISTS Publication.Journal(journalId INTEGER not NULL AUTO_INCREMENT,paperTitle VARCHAR(255),year INTEGER,publicationChannel VARCHAR(255),journalName VARCHAR(255),volume INTEGER,page VARCHAR(255),PRIMARY KEY (journalId));
CREATE TABLE IF NOT EXISTS Publication.Relation(relId INTEGER not NULL AUTO_INCREMENT,paperTitle VARCHAR(255),authorName VARCHAR(255),PRIMARY KEY (relId));
CREATE TABLE IF NOT EXISTS Publication.JournalGeo(geoId INTEGER NOT NULL AUTO_INCREMENT,g GEOMETRY NOT NULL,SPATIAL INDEX(g),paperID INTEGER, PRIMARY KEY (geoId)) ENGINE=MyISAM;
